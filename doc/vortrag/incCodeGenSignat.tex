\frametitle{Modulsignatur}

\begin{itemize}
\item Signatur für semantische Analyse erforderlich
\item Inhalt:
	\begin{itemize}
	\item Importliste
	\item Datendefinitionen
	\item Funktionssignaturen
	\end{itemize}
\item Mögliche Signaturformate:
	\begin{itemize}
	\item native Serialisierung
	\item SL
	\item JSON
	\end{itemize}
\end{itemize}

\end{frame}
